3.3 Bitwise and Advanced Operators

Definition: Bitwise operators (&, |, ^, ~, <<, >>)

Bitwise operators perform operations on the binary representations of numbers.

Why Learn Bitwise Operators?
  • Used in low-level programming, graphics, cryptography, and performance optimization
  • Essential for understanding how computers work at the hardware level
  • Can create highly efficient algorithms
  • Used in flags and permissions systems
The Six Bitwise Operators:
Operator Name Description Example Binary Result Decimal Result
& AND 1 only if both bits are 1 5 & 3 101 & 011 1
| OR 1 if at least one bit is 1 5 | 3 101 | 011 7
^ XOR 1 if bits are different 5 ^ 3 101 ^ 011 6
~ NOT Inverts all bits ~5 ~0101 -6
<< Left Shift Shift bits left, fill with 0s 5 << 1 101 << 1 10
>> Right Shift (Signed) Shift bits right, preserve sign 10 >> 1 1010 >> 1 5
Visualizing Binary Operations:
// 5 in binary: 00000101 // 3 in binary: 00000011 // --------------- // 5 & 3: 00000001 ‚Üí 1 // 5 | 3: 00000111 ‚Üí 7 // 5 ^ 3: 00000110 ‚Üí 6 // ~5: 11111010 ‚Üí -6 (two's complement) // 5 << 1: 00001010 ‚Üí 10 // 10 >> 1: 00000101 ‚Üí 5
Practical Examples:
// 1. Checking if a number is even or odd
function isEven(num) {
    return (num & 1) === 0; // Check if last bit is 0
}

console.log(isEven(4)); // true (binary: 100, last bit 0)
console.log(isEven(7)); // false (binary: 111, last bit 1)

// 2. Setting/Unsetting flags (permissions system)
const PERMISSIONS = {
    READ: 1,     // 0001
    WRITE: 2,    // 0010
    EXECUTE: 4,  // 0100
    DELETE: 8    // 1000
};

let userPermissions = PERMISSIONS.READ | PERMISSIONS.WRITE; // 0011

// Check if user has read permission
console.log((userPermissions & PERMISSIONS.READ) !== 0); // true

// Add execute permission
userPermissions |= PERMISSIONS.EXECUTE; // 0111
console.log(userPermissions); // 7

// Remove write permission
userPermissions &= ~PERMISSIONS.WRITE; // 0101
console.log(userPermissions); // 5

// 3. Fast multiplication/division by powers of 2
function multiplyByTwo(n) {
    return n << 1; // Same as n * 2
}

function divideByTwo(n) {
    return n >> 1; // Same as Math.floor(n / 2)
}

console.log(multiplyByTwo(5)); // 10
console.log(divideByTwo(7)); // 3 (not 3.5!)

// 4. Swapping two variables without temp variable
let a = 5, b = 10;
a ^= b; // a = 5 ^ 10 = 15
b ^= a; // b = 10 ^ 15 = 5
a ^= b; // a = 15 ^ 5 = 10
console.log(a, b); // 10, 5

// 5. Converting to integer (fast truncation)
let num = 3.14159;
let intNum = num | 0; // Truncates decimal part
console.log(intNum); // 3

// 6. Creating unique IDs with bitwise OR
function generateId() {
    return Date.now() | Math.random() * 1000000;
}
console.log(generateId()); // Large number
üí° Pro Tip: Bitwise operators are faster than arithmetic operators for certain operations because they work directly on binary representation!
Definition: Ternary operator (condition ? true : false)

The ternary operator is a concise way to write simple if-else statements.

Syntax:
// Basic syntax
condition ? valueIfTrue : valueIfFalse;
Simple Examples:
// Traditional if-else
let age = 20;
let status;
if (age >= 18) {
    status = "Adult";
} else {
    status = "Minor";
}

// Ternary equivalent
let status2 = age >= 18 ? "Adult" : "Minor";

// More examples
let temperature = 25;
let weather = temperature > 30 ? "Hot" : temperature > 20 ? "Warm" : "Cool";

let score = 85;
let grade = score >= 90 ? "A" : score >= 80 ? "B" : score >= 70 ? "C" : "D";

let userName = "";
let displayName = userName ? userName : "Guest";
// Or better yet:
let displayName2 = userName || "Guest"; // Simpler for truthy/falsy checks

// Conditional rendering in templates
let isLoggedIn = true;
let greeting = `Hello, ${isLoggedIn ? "User" : "Guest"}!`;

// Function returning different values
function getDiscount(price, isMember) {
    return isMember ? price * 0.9 : price;
}
Nested Ternaries (Be Careful!):
// Nested ternary - hard to read!
let result = condition1 ? value1 : condition2 ? value2 : condition3 ? value3 : defaultValue;

// Equivalent if-else chain:
let result2;
if (condition1) {
    result2 = value1;
} else if (condition2) {
    result2 = value2;
} else if (condition3) {
    result2 = value3;
} else {
    result2 = defaultValue;
}

// Better approach: Use switch or if-else for complex logic
function determineCategory(score) {
    if (score >= 90) return "Excellent";
    if (score >= 80) return "Good";
    if (score >= 70) return "Average";
    if (score >= 60) return "Below Average";
    return "Poor";
}
Practical Applications:

// 2. Dynamic styling

// 5. Array manipulation

Definition: delete operator

The delete operator removes a property from an object.

Syntax and Usage:
// Delete object properties
let user = {
    name: "Alice",
    age: 25,
    email: "alice@example.com",
    isAdmin: true
};

console.log(user); // {name: "Alice", age: 25, email: "alice@example.com", isAdmin: true}

delete user.age;
delete user.isAdmin;

console.log(user); // {name: "Alice", email: "alice@example.com"}

// Delete array elements (creates holes!)
let numbers = [1, 2, 3, 4, 5];
delete numbers[2]; // Deletes index 2
console.log(numbers); // [1, 2, empty, 4, 5]
console.log(numbers.length); // 5 (length unchanged!)
console.log(numbers[2]); // undefined

// Can't delete variables declared with let/const
let myVar = 10;
delete myVar; // false (no effect)
console.log(myVar); // 10

const MY_CONST = 42;
delete MY_CONST; // false (no effect)
console.log(MY_CONST); // 42

// Can delete global variables (in browser)
window.myGlobal = "test";
delete window.myGlobal; // true
console.log(window.myGlobal); // undefined

// Return value: true/false
let obj = {prop: 1};
console.log(delete obj.prop); // true (property deleted)
console.log(delete obj.nonexistent); // true (nonexistent property - no error!)
console.log(delete 10); // true (can't delete literals, but doesn't throw error)

// Strict mode behavior
'use strict';
let strictObj = {prop: 1};
delete strictObj.prop; // true
delete strictObj.nonexistent; // true
// delete myVar; // SyntaxError in strict mode
Important Limitations:
  • Cannot delete variables declared with var, let, or const
  • Cannot delete functions declared with function keyword
  • Cannot delete properties marked as non-configurable
  • Doesn't work reliably with host objects (browser APIs)
Practical Example: Cleaning Up Objects
// API response cleanup
function cleanApiResponse(response) {
    const cleaned = {...response}; // Create shallow copy
    
    // Remove sensitive information
    delete cleaned.password;
    delete cleaned.secretKey;
    delete cleaned.token;
    
    // Remove empty/null fields
    Object.keys(cleaned).forEach(key => {
        if (cleaned[key] === null || cleaned[key] === undefined) {
            delete cleaned[key];
        }
    });
    
    return cleaned;
}

let apiData = {
    userId: 123,
    username: "alice",
    password: "secret123",
    secretKey: "abc123",
    token: "xyz789",
    profile: null,
    preferences: undefined,
    createdAt: "2023-01-01"
};

let cleanData = cleanApiResponse(apiData);
console.log(cleanData);
// {userId: 123, username: "alice", createdAt: "2023-01-01"}

Definition: void operator

The void operator evaluates an expression and returns undefined.

Syntax:
void expression
Basic Usage:
// Always returns undefined
console.log(void 0); // undefined
console.log(void 1); // undefined
console.log(void "hello"); // undefined
console.log(void true); // undefined
console.log(void false); // undefined
console.log(void null); // undefined
console.log(void undefined); // undefined

// With function calls
function doSomething() {
    console.log("Doing something...");
    return "result";
}

let result1 = doSomething(); // Logs "Doing something..." and returns "result"
let result2 = void doSomething(); // Logs "Doing something..." and returns undefined

console.log(result1); // "result"
console.log(result2); // undefined

// In URLs (legacy use)
// <a href="javascript:void(0)">Click me</a>
// Prevents page navigation while allowing JS execution

// Modern alternative: event.preventDefault()
document.getElementById('myLink').addEventListener('click', function(e) {
    e.preventDefault();
    // Do something
});
Practical Applications:
1. Ensuring undefined return value
// IIFE (Immediately Invoked Function Expression)
let result = (function() {
    // Do some setup
    console.log("Initializing...");
    return 42;
})();

// If we want to ensure it returns undefined:
let result2 = void (function() {
    console.log("Initializing...");
    return 42;
})(); // returns undefined, not 42
2. In arrow functions
// Arrow functions with blocks return undefined implicitly
let fn1 = () => {
    console.log("Arrow function");
    // No return statement means undefined
};

// Explicitly make it return undefined
let fn2 = () => void console.log("Arrow function");

// Both behave the same:
console.log(fn1()); // undefined
console.log(fn2()); // undefined
3. TypeScript/Flow annotations
// In type systems, void indicates a function that doesn't return anything meaningful
function logAndReturnVoid(message) {
    console.log(message);
    return void 0; // Explicitly return undefined
}

// Used for functions that have side effects but don't return useful data
function handleClick() {
    void performAction(); // Indicates we're calling for side effects only
    void updateUI(); // We don't care about return value
}
4. Preventing return values in expressions
// In minified code or complex expressions
// Without void:
let x = (function() { 
    console.log("Side effect");
    return "value";
})();

// With void - ensures no return value interferes

    console.log("Side effect");
    return "value";
})();

// Useful when you need to use a function call as an expression
// but don't want its return value to affect the overall expression

3.4 Operator Precedence and Associativity

Definition: Understanding operator precedence

Operator precedence determines the order in which operators are evaluated in an expression.

Precedence Levels (Highest to Lowest):
Level Operators Associativity Examples
1 () [] . ?. Left-to-right a.b(), a[b]
2 ++ -- ! ~ + - typeof void delete Right-to-left ++x, !a, typeof x
3 ** Right-to-left 2 ** 3 ** 2
4 * / % Left-to-right 6 / 2 * 3
5 + - Left-to-right 10 - 5 + 3
6 << >> >>> Left-to-right 8 << 2 >> 1
7 < <= > >= instanceof Left-to-right 5 < 10 && 10 < 20
8 == != === !== Left-to-right 5 == 5 && 5 === 5
9 & Left-to-right 5 & 3 & 1
10 ^ Left-to-right 5 ^ 3 ^ 1
11 | Left-to-right 5 | 3 | 1
12 && Left-to-right a && b && c
13 || Left-to-right a || b || c
14 ??: Right-to-left a ?? b ?? c
15 = += -= *= etc. Right-to-left a = b = c
16 , Left-to-right (a, b, c)
Real-world Examples:
// Example 1: Arithmetic precedence
let result1 = 5 + 3 * 2; // 11, not 16
// Multiplication (*) has higher precedence than addition (+)
// So: 5 + (3 * 2) = 5 + 6 = 11

let result2 = (5 + 3) * 2; // 16
// Parentheses override precedence

// Example 2: Logical operator precedence
let a = true, b = false, c = true;
let result3 = a || b && c; // true
// AND has higher precedence than OR
// So: a || (b && c) = true || (false && true) = true || false = true

let result4 = (a || b) && c; // true
// Different grouping gives different result

// Example 3: Assignment precedence
let x = 1, y = 2, z = 3;
x = y = z + 1; // x = 4, y = 4, z = 3
// Addition first: z + 1 = 4
// Then right-to-left assignment: y = 4, then x = 4

// Example 4: Complex expression
let value = 5;
let result5 = ++value * 2 + 3 - 1;
// Step-by-step:
// 1. ++value ‚Üí value becomes 6, returns 6
// 2. 6 * 2 ‚Üí 12
// 3. 12 + 3 ‚Üí 15
// 4. 15 - 1 ‚Üí 14
console.log(result5); // 14
üí° Pro Tip: When in doubt, use parentheses! They make your code more readable and eliminate ambiguity.
Definition: Left-to-right vs right-to-left associativity

Associativity determines the order of evaluation when operators have the same precedence.

Left-to-Right Associativity:

Operators are evaluated from left to right.

// Left-to-right: Addition and subtraction
let result1 = 10 - 5 - 2; // 3
// Evaluated as: (10 - 5) - 2 = 5 - 2 = 3

// Left-to-right: Multiplication and division
let result2 = 12 / 3 / 2; // 2
// Evaluated as: (12 / 3) / 2 = 4 / 2 = 2

// Left-to-right: Logical AND
let a = true, b = true, c = false;
let result3 = a && b && c; // false
// Evaluated as: (a && b) && c = (true && true) && false = true && false = false

// Left-to-right: Bitwise AND
let result4 = 7 & 3 & 1; // 1
// Evaluated as: (7 & 3) & 1 = 3 & 1 = 1
Right-to-Left Associativity:

Operators are evaluated from right to left.

// Right-to-left: Exponentiation
let result5 = 2 ** 3 ** 2; // 512
// Evaluated as: 2 ** (3 ** 2) = 2 ** 9 = 512
// NOT: (2 ** 3) ** 2 = 8 ** 2 = 64

// Right-to-left: Assignment
let x, y, z;
x = y = z = 5; // All become 5
// Evaluated as: x = (y = (z = 5))
// Step 1: z = 5
// Step 2: y = 5
// Step 3: x = 5

// Right-to-left: Ternary operator
let a = true, b = false, c = true;
let result6 = a ? 1 : b ? 2 : c ? 3 : 4;
// Evaluated as: a ? 1 : (b ? 2 : (c ? 3 : 4))
// Since a is true, result is 1

// Right-to-left: Void operator
let result7 = void a = 5; // undefined
// Evaluated as: void (a = 5)
// First: a = 5
// Second: void 5 ‚Üí undefined
Why Associativity Matters:
// Critical example: Exponentiation
// Without knowing right-to-left associativity:
console.log(2 ** 3 ** 2); // What do you expect?

// Many think: (2 ** 3) ** 2 = 8 ** 2 = 64
// But actual: 2 ** (3 ** 2) = 2 ** 9 = 512

// This matters in scientific calculations!
let base = 2;
let exponent = 3;
let power = 2;

// Correct mathematical notation: base^(exponent^power)
let correctResult = base ** (exponent ** power); // 512
// Wrong: (base ** exponent) ** power = 64

// Another example: Assignment chaining
let user = {};
let config = {};

// This assigns the same object to both variables
user.settings = config.settings = {theme: "dark"};
// Same as: user.settings = (config.settings = {theme: "dark"})

console.log(user.settings === config.settings); // true
// Both point to the same object!

// If you wanted separate objects:
user.settings = {theme: "dark"};
config.settings = {theme: "dark"};
console.log(user.settings === config.settings); // false
üéØ Key Insight: Right-to-left associativity is crucial for exponentiation and assignment operations. Misunderstanding this leads to serious bugs!
Definition: Grouping with parentheses

Parantheses override default precedence and associativity rules.

Force Evaluation Order:
// Without parentheses - default precedence
let result1 = 5 + 3 * 2; // 11 (multiplication first)

// With parentheses - force addition first
let result2 = (5 + 3) * 2; // 16

// Complex expression with multiple groupings
let result3 = ((a + b) * c) / (d - e);

// Multiple levels of nesting
let result4 = (((a + b) * c) - d) / (e + f);

// Practical example: Financial calculation
let principal = 1000;
let rate = 0.05;
let years = 10;
let compoundFrequency = 12;

// Simple interest
let simpleInterest = principal * rate * years;

// Compound interest with proper grouping
let compoundInterest = principal * Math.pow(1 + rate / compoundFrequency, compoundFrequency * years);

// Without parentheses - wrong!
let wrongCompound = principal * Math.pow(1 + rate / compoundFrequency, compoundFrequency * years);
// This is actually correct due to precedence, but...

// Even clearer with explicit grouping:
let clearCompound = principal * Math.pow(
    (1 + (rate / compoundFrequency)), 
    (compoundFrequency * years)
);

// String concatenation with numbers
let message = "Total: " + 10 + 5; // "Total: 105"
let correctMessage = "Total: " + (10 + 5); // "Total: 15"
Improving Readability:
// Complex conditional - hard to read
if (user && user.profile && user.profile.age >= 18 && user.profile.isActive) {
    // Do something
}

// Better with parentheses for clarity
if (
    user && 
    user.profile && 
    (user.profile.age >= 18) && 
    user.profile.isActive
) {
    // Do something
}

// Complex boolean logic
let canAccess = 
    (isAdmin || isManager) && 
    (hasLicense || hasPermission) && 
    (!isBlocked);

// Using parentheses makes the logic crystal clear
// You can see exactly what conditions must be met
Workaround for Operator Precedence Issues:
// Problem: Trying to use logical operators with assignment
let flag = true;
let result = flag && (flag = false); // Unexpected behavior!

// Better approach: Split into separate statements
let flag = true;
let temp = flag;
flag = false;
let result = temp;

// Another example: Ternary with assignment
let count = 0;
let result2 = count > 0 ? (count = count - 1) : 0;
// This works but is confusing

// Clearer approach:
let count = 0;
let newValue;
if (count > 0) {
    count -= 1;
    newValue = count;
} else {
    newValue = 0;
}
let result2 = newValue;
üí° Rule of Thumb: When in doubt, add parentheses. It's better to be slightly verbose than ambiguous!
Definition: Common precedence gotchas

Here are the most common pitfalls developers encounter with operator precedence.

Gotcha 1: Logical AND/OR with Comparison Operators
// ‚ùå Common mistake
let age = 25;
if (age >= 18 && age <= 65) { // Correct
    console.log("Adult");
}

// ‚ùå Mistake: Missing parentheses
if (age >= 18 && age <= 65) { // Actually correct!
    // This works fine
}

// ‚ùå Real gotcha: Confusing precedence
let isValid = false;
let result1 = isValid && age >= 18 || age <= 65; // Unexpected result!
// This is evaluated as: (isValid && age >= 18) || age <= 65
// Since isValid is false, it becomes: false || age <= 65
// Which is true if age <= 65, regardless of the first condition!

// ‚úÖ Fix: Use explicit grouping
let result2 = (isValid && age >= 18) || (age <= 65); // Clear intent
// Or better yet, restructure logic
Gotcha 2: Assignment in Conditionals
// ‚ùå Common mistake
let x = 5;
if (x = 10) { // Oops! Assignment instead of comparison
    console.log("This always runs!");
}

// Why? Because assignment returns the assigned value
// x = 10 returns 10, which is truthy

// ‚úÖ Fix: Use strict equality
if (x === 10) {
    console.log("Correct comparison");
}

// ‚úÖ Intentional assignment in condition (rare cases)
let user;
while (user = getNextUser()) {
    processUser(user);
}
// This is valid but dangerous - use comments to explain
// while (user = getNextUser()) { // Assign and check
Gotcha 3: Bitwise vs Logical Operators
// ‚ùå Common confusion
let flag1 = true;
let flag2 = false;

// These look similar but are completely different
if (flag1 && flag2) { // Logical AND - returns false
    console.log("Both true");
}

if (flag1 & flag2) { // Bitwise AND - converts to numbers: 1 & 0 = 0
    console.log("Both true"); // This won't run
}

// In real code:
let permissions = 7; // 111 in binary (read=1, write=2, execute=4)
if (permissions & 1) { // Check read permission
    console.log("Can read");
}

// Don't mix them up!
Gotcha 4: Nullish Coalescing vs Logical OR
// ‚ùå Common mistake
let config = {
    timeout: 0, // User wants 0ms timeout
    retries: "", // User wants no retries
    debug: false // User wants debugging off
};

// Wrong: Using || for defaults
let timeout = config.timeout || 5000; // 5000 (wrong! should be 0)
let retries = config.retries || 3; // 3 (wrong! should be "")
let debug = config.debug || true; // true (wrong! should be false)

// ‚úÖ Correct: Use ?? for nullish coalescing
let timeout2 = config.timeout ?? 5000; // 0 (correct!)
let retries2 = config.retries ?? 3; // "" (correct!)
let debug2 = config.debug ?? true; // false (correct!)
Gotcha 5: Exponentiation and Negative Numbers
// ‚ùå Common mistake
let x = -2;
let square = x ** 2; // 4 (correct)

// But what about this?
let result = -2 ** 2; // -4 (not 4!)

// Why? Because exponentiation has higher precedence than unary minus
// So: -(2 ** 2) = -(4) = -4

// ‚úÖ Fix: Use parentheses
let correctResult = (-2) ** 2; // 4
let anotherCorrect = Math.pow(-2, 2); // 4

// Another example:
let a = 3;
let b = -2;
let result2 = a * b ** 2; // 3 * (-2 ** 2) = 3 * -4 = -12 (wrong!)
let correctResult2 = a * (b ** 2); // 3 * 4 = 12 (correct!)
Gotcha 6: Template Literals with Expressions
// ‚ùå Common mistake
let price = 10;
let taxRate = 0.08;
let total = `${price + taxRate}`; // "10.08" (wrong!)

// Expected: Total price including tax
// Actual: Just adding price and tax rate

// ‚úÖ Fix: Calculate properly
let totalWithTax = `${price * (1 + taxRate)}`; // "10.8"

// Even better: Format currency
let formattedTotal = `$${(price * (1 + taxRate)).toFixed(2)}`; // "$10.80"

// Another example:

let badGreeting = `Hello ${name.toUpperCase}`; // "Hello function toUpperCase() { ... }" (WRONG!)
