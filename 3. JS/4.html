3.1 Arithmetic and Assignment Operators

Definition: Basic arithmetic (+, -, *, /, %)

Arithmetic operators perform mathematical operations on numbers.

The Five Fundamental Operators:
Operator Name Example Result
+ Addition 5 + 3 8
- Subtraction 10 - 4 6
* Multiplication 7 * 6 42
/ Division 15 / 3 5
% Modulus (Remainder) 17 % 5 2
Practical Examples:
// Basic calculations
let price = 29.99;
let taxRate = 0.08;
let quantity = 3;

let subtotal = price * quantity;           // 89.97
let tax = subtotal * taxRate;              // 7.1976
let total = subtotal + tax;                // 97.1676

console.log(`Subtotal: $${subtotal.toFixed(2)}`);
console.log(`Tax: $${tax.toFixed(2)}`);
console.log(`Total: $${total.toFixed(2)}`);

// Modulus for even/odd detection
function isEven(num) {
    return num % 2 === 0;
}

console.log(isEven(4));  // true
console.log(isEven(7));  // false
console.log(isEven(0));  // true

// Modulus for cycling through arrays
const colors = ['red', 'green', 'blue'];
for (let i = 0; i < 10; i++) {
    console.log(`Item ${i + 1}: ${colors[i % colors.length]}`);
}
// Output: Item 1: red, Item 2: green, Item 3: blue, Item 4: red, etc.

// Modulus for clock arithmetic (hours)
let currentHour = 22;
let futureHour = (currentHour + 5) % 24; // 3 (next day at 3 AM)
console.log(`In 5 hours, it will be ${futureHour}:00`);
üí° Pro Tip: The modulus operator (%) is incredibly useful for wrapping values within ranges - perfect for clocks, calendars, game loops, and array cycling!
Definition: Assignment operators (=, +=, -=, etc.)

Assignment operators assign values to variables. They can also combine assignment with other operations.

Basic Assignment Operator:

Definition: Increment and decrement (++, --)

Increment (++) and decrement (--) operators increase or decrease a variable's value by 1.

Prefix vs Postfix Notation:
Form Name Behavior Example Result
++x Prefix increment Increase first, then return new value let x = 5; y = ++x; y = 6, x = 6
x++ Postfix increment Return original value, then increase let x = 5; y = x++; y = 5, x = 6
--x Prefix decrement Decrease first, then return new value let x = 5; y = --x; y = 4, x = 4
x-- Postfix decrement Return original value, then decrease let x = 5; y = x--; y = 5, x = 4
Visual Demonstration:
// Prefix increment
let a = 10;
let b = ++a; // a becomes 11, then b gets 11
console.log(`a: ${a}, b: ${b}`); // a: 11, b: 11

// Postfix increment
let c = 10;
let d = c++; // d gets 10, then c becomes 11
console.log(`c: ${c}, d: ${d}`); // c: 11, d: 10

// In loops - common pattern
let counter = 0;
while (counter < 5) {
    console.log(counter++);
    // This logs 0, 1, 2, 3, 4
    // counter increments AFTER logging
}

// Prefix in conditionals
let score = 85;
if (++score >= 90) { // score becomes 86, then compare
    console.log("Great job!"); // This runs!
}

// Be careful with multiple operators
let x = 5;
let result = x++ + ++x + x--;
// Step-by-step:
// 1. x++ returns 5, x becomes 6
// 2. ++x makes x 7, returns 7
// 3. x-- returns 7, x becomes 6
// Result: 5 + 7 + 7 = 19
console.log(result); // 19
console.log(x); // 6
When to Use Each:
  • Use postfix (i++): When you need the current value before incrementing (most common in loops)
  • Use prefix (++i): When you need the incremented value immediately
  • Generally avoid: Complex expressions with multiple increment/decrement operators
‚ö†Ô∏è Warning: Avoid using increment/decrement operators in complex expressions. It makes code harder to read and debug!
Definition: Exponentiation operator (**)

The exponentiation operator (**) raises the first operand to the power of the second operand.

Basic Usage:
// Traditional way (Math.pow)
console.log(Math.pow(2, 3)); // 8
console.log(Math.pow(5, 2)); // 25
console.log(Math.pow(10, 0)); // 1
console.log(Math.pow(4, 0.5)); // 2 (square root)

// Modern way with **
console.log(2 ** 3); // 8
console.log(5 ** 2); // 25
console.log(10 ** 0); // 1
console.log(4 ** 0.5); // 2 (square root)
console.log(8 ** (1/3)); // 2 (cube root)

// Negative exponents
console.log(2 ** -2); // 0.25 (1/4)
console.log(10 ** -1); // 0.1 (1/10)

// Fractional exponents
console.log(27 ** (1/3)); // 3 (cube root)
console.log(16 ** 0.25); // 2 (fourth root)
Real-world Applications:
// Calculate area of a square
let side = 5;
let area = side ** 2; // 25
console.log(`Area: ${area}`);

// Calculate volume of a cube
let edge = 3;
let volume = edge ** 3; // 27
console.log(`Volume: ${volume}`);

// Compound interest calculation
let principal = 1000;
let rate = 0.05; // 5%
let years = 10;
let amount = principal * (1 + rate) ** years;
console.log(`Future value: $${amount.toFixed(2)}`); // $1628.89

// Scientific notation
let avogadro = 6.022e23; // 6.022 √ó 10^23
let mass = 12 * avogadro; // Mass of 12g carbon
console.log(mass); // 7.2264e+24

// Factorial function using recursion
function factorial(n) {
    if (n <= 1) return 1;
    return n * factorial(n - 1);
}

// But what if we want to calculate powers?
function power(base, exponent) {
    if (exponent === 0) return 1;
    if (exponent < 0) return 1 / power(base, -exponent);
    return base * power(base, exponent - 1);
}

// Compare performance
console.time('Using **');
let result1 = 2 ** 10;
console.timeEnd('Using **'); // Much faster!

console.time('Using Math.pow');
let result2 = Math.pow(2, 10);
console.timeEnd('Using Math.pow');
Operator Precedence:
// Exponentiation has higher precedence than multiplication/division
console.log(2 * 3 ** 2); // 18 (not 36) because 3**2=9, then 2*9=18
console.log((2 * 3) ** 2); // 36 (because parentheses override)

// Exponentiation is right-associative
console.log(2 ** 3 ** 2); // 512 (not 64)
// This is evaluated as: 2 ** (3 ** 2) = 2 ** 9 = 512
// NOT: (2 ** 3) ** 2 = 8 ** 2 = 64

// Practical implication:
let temperature = 25;
let windSpeed = 10;
// Wind chill formula (simplified)
let windChill = 13.12 + 0.6215 * temperature - 11.37 * (windSpeed ** 0.16) + 0.3965 * temperature * (windSpeed ** 0.16);
console.log(`Wind chill: ${windChill.toFixed(1)}¬∞C`);
üí° Pro Tip: The exponentiation operator is the most powerful math operator in JavaScript - use it for scientific calculations, financial modeling, and any exponential growth scenarios!

3.2 Comparison and Logical Operators

Definition: Equality operators (== vs ===)

JavaScript has two types of equality operators: loose (==) and strict (===).

Strict Equality (===): Type and Value
// Strict equality checks both type AND value
console.log(5 === 5); // true
console.log("5" === "5"); // true
console.log(5 === "5"); // false (number vs string)
console.log(true === true); // true
console.log(true === 1); // false (boolean vs number)
console.log(null === null); // true
console.log(undefined === undefined); // true
console.log(null === undefined); // false
console.log([1,2,3] === [1,2,3]); // false (different objects)
console.log({} === {}); // false (different objects)
Loose Equality (==): Type Coercion
// Loose equality coerces types before comparison
console.log(5 == 5); // true
console.log("5" == 5); // true (string "5" coerced to number 5)
console.log(true == 1); // true (true coerced to 1)
console.log(false == 0); // true (false coerced to 0)
console.log("" == 0); // true (empty string coerced to 0)
console.log(" " == 0); // true (whitespace string coerced to 0)
console.log(null == undefined); // true (special case)
console.log([] == 0); // true (empty array coerced to empty string, then to 0)
console.log([5] == 5); // true (array [5] coerced to "5", then to 5)
console.log([1,2,3] == "1,2,3"); // true (array coerced to string)
Why This Matters - Real Example:
// ‚ùå Dangerous: Using == for form validation
let userInput = prompt("Enter your age:");
if (userInput == 18) {
    console.log("You're an adult!");
} else {
    console.log("You're not an adult yet.");
}

// What happens if user enters "18"? It works!
// What happens if user enters "018"? It still works!
// What happens if user enters ""? It evaluates to false - good!
// What happens if user enters "abc"? It evaluates to false - good!

// BUT... what if user enters "0"?
// "0" == 18 ‚Üí false (good)
// But "0" == 0 ‚Üí true (problem!)

// ‚úÖ Correct approach: Always use ===
let ageInput = prompt("Enter your age:");
if (ageInput !== null && Number(ageInput) === 18) {
    console.log("You're an adult!");
} else {
    console.log("You're not an adult yet.");
}

// Even better: Validate input first
function validateAge(input) {
    const age = Number(input);
    return !isNaN(age) && age >= 0 && age === 18;
}

console.log(validateAge("18")); // true
console.log(validateAge("18.0")); // false (18.0 !== 18)
console.log(validateAge("abc")); // false
console.log(validateAge("")); // false
The Nullish Coalescing Rule:
// Special cases where == behaves unexpectedly
console.log("0" == false); // true
console.log("0" == 0); // true
console.log(false == 0); // true
// So: "0" == 0 == false, but "0" !== false!

// This creates a transitive paradox!
// If A == B and B == C, then A == C should be true
// But here: "0" == false and false == 0, so "0" == 0
// However: "0" !== false

// Best practice: Never rely on these rules!
// Use === always, and convert types explicitly when needed
üö® Ultimate Rule: ALWAYS use === and !==. Only use == and != if you specifically need type coercion, and even then, be explicit about your intentions.
Definition: Comparison operators (<, >, <=, >=)

Comparison operators compare two values and return a boolean result.

The Four Comparison Operators:
Operator Name Description Example Result
< Less than Returns true if left operand is less than right 3 < 5 true
> Greater than Returns true if left operand is greater than right 7 > 2 true
<= Less than or equal Returns true if left operand is less than or equal to right 5 <= 5 true
>= Greater than or equal Returns true if left operand is greater than or equal to right 10 >= 8 true
String Comparison:
// Strings are compared lexicographically (dictionary order)
console.log("apple" < "banana"); // true
console.log("zebra" > "apple"); // true
console.log("Apple" < "apple"); // true (uppercase comes before lowercase in ASCII)
console.log("10" < "2"); // true! ("1" comes before "2" in string comparison)

// Why this happens:
// String comparison compares character by character using Unicode values
// "1" (Unicode 49) vs "2" (Unicode 50) ‚Üí "1" < "2" ‚Üí true

// To fix numeric string comparison:
function compareNumbers(a, b) {
    return Number(a) - Number(b);
}

console.log(compareNumbers("10", "2")); // 8 (positive means 10 > 2)
console.log(Number("10") < Number("2")); // false

// Date comparison
let date1 = new Date("2023-01-15");
let date2 = new Date("2023-02-20");
console.log(date1 < date2); // true
console.log(date1.getTime() < date2.getTime()); // true (same thing)

// Object comparison (reference comparison)
let obj1 = {value: 5};
let obj2 = {value: 5};
let obj3 = obj1;

console.log(obj1 < obj2); // false (objects can't be compared like this)
console.log(obj1 === obj3); // true (same reference)
console.log(obj1 === obj2); // false (different objects)

Definition: Logical operators (&&, ||, !)

Logical operators work with boolean values and return boolean results.

The Three Logical Operators:
Operator Name Description Truth Table
&& AND Returns true only if both operands are truthy true && true ‚Üí true
true && false ‚Üí false
false && true ‚Üí false
false && false ‚Üí false
|| OR Returns true if at least one operand is truthy true || true ‚Üí true
true || false ‚Üí true
false || true ‚Üí true
false || false ‚Üí false
! NOT Reverses the boolean value of its operand !true ‚Üí false
!false ‚Üí true
Short-Circuit Evaluation:

JavaScript uses short-circuit evaluation with logical operators:

  • AND (&&): If first operand is falsy, returns it without evaluating second
  • OR (||): If first operand is truthy, returns it without evaluating second
// AND short-circuit
console.log(false && someFunction()); // false (someFunction never called)
console.log(true && someFunction()); // calls someFunction and returns its value

// OR short-circuit
console.log(true || someFunction()); // true (someFunction never called)
console.log(false || someFunction()); // calls someFunction and returns its value

// Practical examples:
function someFunction() {
    console.log("This function was called!");
    return "result";
}

// Safe property access
let user = {profile: {name: "Alice"}};

// Before optional chaining (ES2020):
let userName = user && user.profile && user.profile.name;
console.log(userName); // "Alice"

// Now with optional chaining:
userName = user?.profile?.name;
console.log(userName); // "Alice"

// Default values with OR
let username = "";
let displayName = username || "Guest";
console.log(displayName); // "Guest"

// Multiple defaults
let theme = process.env.THEME || "light" || "dark";
console.log(theme); // "light" (first truthy value)

Definition: Nullish coalescing operator (??)

The nullish coalescing operator (??) returns its right-hand operand when the left-hand operand is null or undefined.

Traditional OR vs Nullish Coalescing:
Expression With || With ?? Explanation
null || "default" "default" "default" Same behavior
undefined || "default" "default" "default" Same behavior
"" || "default" "default" "" Different! Empty string is falsy but not nullish
0 || "default" "default" 0 Different! Zero is falsy but not nullish
false || "default" "default" false Different! False is falsy but not nullish
"hello" || "default" "hello" "hello" Same behavior
Practical Examples:
// Configuration defaults

Definition: Optional chaining operator (?.)

The optional chaining operator (?.) allows safe access to deeply nested properties without throwing errors.

Before Optional Chaining (The Old Way):
// ‚ùå Dangerous: Can throw TypeError
let user = {
    profile: {
        address: {
            city: "New York"
        }
    }
};

// This works fine
console.log(user.profile.address.city); // "New York"

// But what if user is null?
let user2 = null;
// console.log(user2.profile.address.city); // TypeError: Cannot read property 'profile' of null

// ‚ùå Workaround: Long chains of checks
let city = user2 && user2.profile && user2.profile.address && user2.profile.address.city;
console.log(city); // undefined

// ‚ùå Even worse: Function calls
function getCity(user) {
    if (user && user.profile && user.profile.address && user.profile.address.city) {
        return user.profile.address.city;
    }
    return "Unknown";
}
With Optional Chaining (Modern Approach):
// ‚úÖ Modern and clean
let user = {
    profile: {
        address: {
            city: "New York"
        }
    }
};

// Safe access - returns undefined if any part is null/undefined
console.log(user?.profile?.address?.city); // "New York"

let user2 = null;
console.log(user2?.profile?.address?.city); // undefined (no error!)

// Safe method calls
let user3 = {
    getName() {
        return "Alice";
    }
};

let user4 = null;
console.log(user3?.getName()); // "Alice"
console.log(user4?.getName()); // undefined

// Safe array indexing
let users = [
    {name: "Alice"},
    {name: "Bob"}
];

let user5 = null;
console.log(users?.[0]?.name); // "Alice"
console.log(user5?.[0]?.name); // undefined

// Safe dynamic property access
let prop = "address";
console.log(user?.[prop]?.city); // "New York"
