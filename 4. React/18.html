12.5 Tailwind CSS with React

üìå Summary: Utility-First CSS at Lightning Speed

Tailwind CSS is a utility-first CSS framework that lets you build designs directly in your HTML/JSX using small, single-purpose classes. No custom CSS files needed!

üîπ Core Philosophy

  • Utility-first: Instead of writing CSS, compose styles from pre-defined utility classes
  • Responsive by default: Breakpoints built into every utility
  • No naming conflicts: Classes are scoped to your design system
  • Highly customizable: Configure colors, spacing, fonts via tailwind.config.js

Responsive design means your app looks great on phones, tablets, laptops, and TVs. It's not optional ‚Äî it's essential.

üîπ The Mobile-First Approach

üî• Hook-Based Responsive Detection

Create a custom hook to detect screen size in JavaScript:


import { useState, useEffect } from 'react';

function useResponsive() {
  const [size, setSize] = useState('mobile');

  useEffect(() => {
    const checkSize = () => {
      if (window.innerWidth < 640) {
        setSize('mobile');
      } else if (window.innerWidth < 1024) {
        setSize('tablet');
      } else {
        setSize('desktop');
      }
    };

    // Initial check
    checkSize();
    
    // Listen for resize
    window.addEventListener('resize', checkSize);
    
    return () => window.removeEventListener('resize', checkSize);
  }, []);

  return size;
}

// Usage
function MyComponent() {
  const screen = useResponsive();
  
  return (
    <div>
      {screen === 'mobile' && <p>Mobile view</p>}
      {screen === 'tablet' && <p>Tablet view</p>}
      {screen === 'desktop' && <p>Desktop view</p>}
    </div>
  );
}

‚ö° React Context Theme Provider


import { createContext, useContext, useState } from 'react';

const ThemeContext = createContext();

export function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('dark');

  const toggleTheme = () => {
    setTheme(prev => prev === 'dark' ? 'light' : 'dark');
  };

  const currentTheme = themes[theme];

  return (
    <ThemeContext.Provider value={{ theme, toggleTheme, currentTheme }}>
      {children}
    </ThemeContext.Provider>
  );
}

export function useTheme() {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
}

‚ö° JavaScript Theme Switcher


function ThemeSwitcher() {
  const toggleTheme = () => {
    document.documentElement.classList.toggle('dark-theme');
    document.documentElement.classList.toggle('light-theme');
    
    // Persist preference
    const isDark = document.documentElement.classList.contains('dark-theme');
    localStorage.setItem('theme', isDark ? 'dark' : 'light');
  };

  // Apply saved theme on load
  useEffect(() => {
    const savedTheme = localStorage.getItem('theme');
    if (savedTheme === 'dark') {
      document.documentElement.classList.add('dark-theme');
    } else {
      document.documentElement.classList.add('light-theme');
    }
  }, []);

  return (
    <button onClick={toggleTheme} className="theme-toggle">
      Toggle Theme
    </button>
  );
}

// Usage in components
function Button() {
  return (
    <button style={{
      backgroundColor: 'var(--primary-color)',
      color: 'var(--text-color)',
      border: 'none',
      padding: '12px 24px',
      borderRadius: '6px',
      cursor: 'pointer'
    }}>
      Click Me
    </button>
  );
}

import { animated, useSpring } from '@react-spring/web';

function AnimatedCard() {
  const props = useSpring({
    from: { opacity: 0, transform: 'translateY(20px)' },
    to: { opacity: 1, transform: 'translateY(0)' },
    config: { tension: 200, friction: 20 }
  });

  return (
    <animated.div style={props} className="animated-card">
      <h3>Spring Animation</h3>
      <p>Feels like a real spring!</p>
    </animated.div>
  );
}

// Complex multi-property animation
